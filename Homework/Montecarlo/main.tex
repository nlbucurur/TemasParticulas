\NewDocumentCommand{\braket}{s m o o}{
  \IfBooleanTF{#1}{
    \IfValueTF{#3}{
      \left\langle\left.#2\right|#3\right\rangle
    }{
      \left\langle\left.#2\right|#2\right\rangle
    }
  }{
    \IfValueTF{#3}{
      \IfValueTF{#4}{
        \left\langle#3\left|#2\right|#4\right\rangle
      }{
        \left\langle#3\left|#2\right|#3\right\rangle
      }
    }{
      \left\langle #2\right\rangle
    }
  }
}

%% Comandos para las derivadas ordinarias
\NewDocumentCommand{\diff}{ s m m o }{
  \IfBooleanTF{#1}{
    \IfValueTF{#4}{
      \frac{d^{#4}#2}{d#3^{#4}}
    }{
      \frac{d#2}{d#3}
    }
  }{
    \IfValueTF{#4}{
      \frac{d^{#4}}{d#3^{#4}} #2
    }{
      \frac{d}{d#3} #2
    }
  }
}


%% Comandos para las derivadas parciales
\NewDocumentCommand{\parcial}{ s m m o }{
  \IfBooleanTF{#1}{
    \IfValueTF{#4}{
      \frac{\partial^{#4}}{\partial#3^{#4}} #2
    }{
      \frac{\partial}{\partial#3} #2
    }
  }{
    \IfValueTF{#4}{
      \frac{\partial^{#4}#2}{\partial#3^{#4}}
    }{
      \frac{\partial#2}{\partial#3}
    }
  }
}
